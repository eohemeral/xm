### 5.2.2 路由鉴权
通过Vue Router守卫实现路由鉴权，未登录用户只能访问公开页面。

**关键代码**：
```javascript
// vue/src/router/index.js
import Vue from 'vue'
import Router from 'vue-router'
import Home from '@/views/Home.vue'
import Login from '@/views/Login.vue'
import Register from '@/views/Register.vue'

Vue.use(Router)

const router = new Router({
  routes: [
    {
      path: '/',
      name: 'Home',
      component: Home
    },
    {
      path: '/login',
      name: 'Login',
      component: Login
    },
    {
      path: '/register',
      name: 'Register',
      component: Register
    },
    {
      path: '/user',
      name: 'UserCenter',
      component: () => import('@/views/user/UserCenter.vue'),
      meta: { requiresAuth: true },
      children: [
        {
          path: 'info',
          name: 'UserInfo',
          component: () => import('@/views/user/UserInfo.vue')
        },
        {
          path: 'courses',
          name: 'MyCourses',
          component: () => import('@/views/user/MyCourses.vue')
        },
        {
          path: 'posts',
          name: 'MyPosts',
          component: () => import('@/views/user/MyPosts.vue')
        },
        {
          path: 'comments',
          name: 'MyComments',
          component: () => import('@/views/user/MyComments.vue')
        },
        {
          path: 'collections',
          name: 'MyCollections',
          component: () => import('@/views/user/MyCollections.vue')
        }
      ]
    }
  ]
})

// 路由守卫
router.beforeEach((to, from, next) => {
  const token = localStorage.getItem('token')
  if (to.matched.some(record => record.meta.requiresAuth)) {
    if (!token) {
      next({ name: 'Login' })
    } else {
      next()
    }
  } else {
    next()
  }
})

export default router
```

### 5.2.3 Axios封装
封装axios实例，请求拦截器附加JWT，响应拦截器统一处理错误码。

**关键代码**：
```javascript
// vue/src/utils/request.js
import axios from 'axios'

const request = axios.create({
  baseURL: '/api',
  timeout: 10000
})

// 请求拦截器
request.interceptors.request.use(
  config => {
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.token = token
    }
    return config
  },
  error => {
    return Promise.reject(error)
  }
)

// 响应拦截器
request.interceptors.response.use(
  response => {
    const res = response.data
    if (res.code !== 200) {
      // 统一错误处理
      console.error(res.message)
      return Promise.reject(res)
    }
    return res
  },
  error => {
    console.error('请求失败:', error)
    return Promise.reject(error)
  }
)

export default request
```

```vue
<!-- vue/src/views/course/Detail.vue -->
<template>
  <div class="course-detail">
    <div class="course-header">
      <img :src="course.image" alt="课程图片">
      <div class="course-info">
        <h1>{{ course.title }}</h1>

        <div class="course-meta">
          <span class="category">{{ course.category }}</span>

          </el-button>
          <el-button @click="toggleCollect" v-if="token">
            <i :class="isCollected ? 'el-icon-star-on' : 'el-icon-star-off'"></i>
            {{ isCollected ? '已收藏' : '收藏' }}
          </el-button>
        </div>
      </div>
    </div>
    <div class="course-content">
      <h2>课程介绍</h2>
      <div v-html="course.content"></div>
    </div>
  </div>
</template>

<script>
import request from '@/utils/request'

export default {
  name: 'CourseDetail',
  data() {
    return {
      course: {},
      token: localStorage.getItem('token'),
      isLiked: false,
      isCollected: false
    }
  },
  created() {
    this.loadCourse()
  },
  methods: {
    loadCourse() {
      const id = this.$route.params.id
      request.get(`/course/detail/${id}`).then(res => {
        this.course = res.data || res
        this.checkStatus()
      })
    },
    checkStatus() {
      if (this.token) {
        const id = this.$route.params.id
        request.get(`/course/checkStatus/${id}`).then(res => {
          this.isLiked = res.liked
          this.isCollected = res.collected
        })
      }
    },
    startLearning() {
      if (!this.token) {
        this.$confirm('您还未登录，是否前往登录？', '提示', {
          confirmButtonText: '确定',
          cancelButtonText: '取消',
          type: 'warning'
        }).then(() => {
          this.$router.push('/login')
        })
      } else {
        // 跳转到学习页面
        this.$router.push(`/course/learn/${this.course.id}`)
      }
    },
    toggleLike() {
      const id = this.$route.params.id
      if (this.isLiked) {
        request.post(`/course/unlike/${id}`).then(res => {
          this.isLiked = false
          this.course.likeCount--
        })
      } else {
        request.post(`/course/like/${id}`).then(res => {
          this.isLiked = true
          this.course.likeCount++
        })
      }
    },
    toggleCollect() {
      const id = this.$route.params.id
      if (this.isCollected) {
        request.post(`/course/uncollect/${id}`).then(res => {
          this.isCollected = false
        })
      } else {
        request.post(`/course/collect/${id}`).then(res => {
          this.isCollected = true
        })
      }
    }
  }
}
</script>
```

```vue
<template>
  <div class="post-detail">

          <div class="comment-meta">
            <span class="comment-time">{{ comment.createTime }}</span>
            <span class="comment-like" @click="likeComment(comment.id)">
              <i class="el-icon-thumb"></i> {{ comment.likeCount }}
            </span>
          </div>
        </div>
      </div>
    </div>
  </div>

<script>
import request from '@/utils/request'

export default {
  name: 'PostDetail',
  data() {
    return {
      post: {},
      comments: [],
      commentContent: '',
      token: localStorage.getItem('token'),
      isLiked: false,
      isCollected: false
    }
  },
  created() {
    this.loadPost()
  },
  methods: {
    loadPost() {
      const id = this.$route.params.id
      request.get(`/community/detail/${id}`).then(res => {
        this.post = res.data || res
        this.comments = this.post.comments || []
        this.checkStatus()
      })
    },
    checkStatus() {
      if (this.token) {
        const id = this.$route.params.id
        request.get(`/community/checkStatus/${id}`).then(res => {
          this.isLiked = res.liked
          this.isCollected = res.collected
        })
      }
    },
    like() {
      const id = this.$route.params.id
      request.post(`/community/like/${id}`).then(res => {
        this.isLiked = true
        this.post.likeCount++
      })
    },
    unlike() {
      const id = this.$route.params.id
      request.post(`/community/unlike/${id}`).then(res => {
        this.isLiked = false
        this.post.likeCount--
      })
    },
    collect() {
      const id = this.$route.params.id
      request.post(`/community/collect/${id}`).then(res => {
        this.isCollected = true
      })
    },
    uncollect() {
      const id = this.$route.params.id
      request.post(`/community/uncollect/${id}`).then(res => {
        this.isCollected = false
      })
    },
    submitComment() {
      if (!this.commentContent) return
      const id = this.$route.params.id
      request.post('/community/comment', {
        postId: id,
        content: this.commentContent
      }).then(res => {
        this.commentContent = ''
        this.loadPost()
      })
    },
    likeComment(commentId) {
      request.post(`/community/comment/like/${commentId}`).then(res => {
        const comment = this.comments.find(c => c.id === commentId)
        if (comment) {
          comment.likeCount++
        }
      })
    }
  }
}
</script>
```

### 5.4.1 安全配置

**Spring Security配置**：
```java
// springboot/src/main/java/com/example/springboot/common/SecurityConfig.java
package com.example.springboot.common;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.web.SecurityFilterChain;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.CorsConfigurationSource;
import org.springframework.web.cors.UrlBasedCorsConfigurationSource;

import java.util.Arrays;

@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain securityFilterChain(HttpSecurity http) throws Exception {
        http
            .cors(cors -> cors.configurationSource(corsConfigurationSource()))
            .csrf(csrf -> csrf.disable())
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/user/login", "/user/register", "/user/info").permitAll()
                .requestMatchers("/course/list", "/course/detail/**", "/course/recommend").permitAll()
                .requestMatchers("/community/list", "/community/detail/**", "/community/stats/**", "/community/admin/**", "/community/myPosts", "/community/myComments", "/community/myCollections", "/community/post", "/community/comment", "/community/like/**", "/community/unlike/**", "/community/collect/**", "/community/uncollect/**", "/community/comment/like/**", "/community/comment/unlike/**").permitAll()
                .anyRequest().authenticated()
            );

        return http.build();
    }

    @Bean
    public CorsConfigurationSource corsConfigurationSource() {
```

## 5.5 个性化推荐实现

### 5.5.1 推荐策略
采用基于物品的简单协同过滤策略，核心实现逻辑如下：

1. **用户行为记录**：
   - 用户浏览课程时，将`view_count`字段加1
   - 学习行为记录至`study_record`表
   - 收藏行为记录至`collection`表

2. **推荐算法**：
   - 获取用户学过或收藏的课程类型
   - 从同类课程中筛选出浏览量最高的Top 10作为推荐内容
   - 针对无历史记录的用户，采用冷启动方案，直接返回全局热门课程

3. **技术特点**：
   - 实时性好：推荐结果能够实时更新
   - 实现简单：全程基于SQL查询即可完成，无需复杂算法
   - 可解释性强：用户能够清晰理解推荐原因
   - 适配性佳：特别适合用户量不大的平台初期阶段使用

### 5.5.2 推荐SQL实现

**推荐课程查询**：
```xml
<!-- springboot/src/main/resources/mapper/CourseMapper.xml -->
<select id="recommend" resultType="com.example.springboot.entity.Course">
    <!-- 有历史记录的用户推荐 -->
    <choose>
        <when test="userId != null">
            SELECT c.* FROM course c
            WHERE c.category IN (
                SELECT DISTINCT co.category FROM course co
                JOIN study_record sr ON co.id = sr.course_id
                WHERE sr.user_id = #{userId}
                UNION
                SELECT DISTINCT cc.category FROM course cc
                JOIN collection col ON cc.id = col.target_id
                WHERE col.user_id = #{userId} AND col.target_type = 'course'
            )
            AND c.id NOT IN (
                SELECT course_id FROM study_record WHERE user_id = #{userId}
                UNION
                SELECT target_id FROM collection WHERE user_id = #{userId} AND target_type = 'course'
            )
            ORDER BY c.view_count DESC
            LIMIT 10
        </when>
        <!-- 冷启动方案 -->
        <otherwise>
            SELECT * FROM course
            ORDER BY view_count DESC
            LIMIT 10
        </otherwise>
    </choose>
</select>
```
